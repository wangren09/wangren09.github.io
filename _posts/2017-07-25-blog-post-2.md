In computer science, a linked list is a linear collection of data elements, in which linear order is not given by their physical placement in memory. Each points to the next node by means of a pointer. It is a data structure consisting of a group of nodes which together represent a sequence. Under the simplest form, each node is composed of data and a reference (in other words, a link) to the next node in the sequence.

Definition for singly-linked list.
<pre>
class ListNode(object):
    def __init__(self, x):
        self.val = x
        self.next = None
</pre>

We often define dummy node and cur node to solve linked list problem.

## 2. Add Two Numbers
<pre>
class Solution(object):
    
    def addTwoNumbers(self, l1, l2):
        """
        :type l1: ListNode
        :type l2: ListNode
        :rtype: ListNode
        """
        dummy = head = ListNode(0)
        carry = 0
        while l1 != None or l2 != None or carry != 0:
            v1 = v2 = 0
            if l1 != None:
                v1 = l1.val
                l1 = l1.next
            if l2 != None:
                v2 = l2.val
                l2 = l2.next
            val = (v1+v2+carry)%10
            carry = (v1+v2+carry)/10
            head.next = ListNode(val)
            head = head.next
        return dummy.next
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 19. Remove Nth Node From End of List
<pre>
class Solution(object):
    
    def removeNthFromEnd(self, head, n):
        """
        :type head: ListNode
        :type n: int
        :rtype: ListNode
        """
        l = r = head
        step = 0
        while step < n:
            r = r.next
            step += 1
        if r == None:
            return head.next
        while r.next != None:
            l = l.next
            r = r.next
        l.next = l.next.next
        return head
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 21. Merge Two Sorted Lists
<pre>
class Solution(object):
    
    def mergeTwoLists(self, l1, l2):
        """
        :type l1: ListNode
        :type l2: ListNode
        :rtype: ListNode
        """
        dummy = ListNode(0)
        cur = dummy
        while l1 != None and l2 != None:
            if l1.val <= l2.val:
                cur.next = l1
                l1 = l1.next
            else:
                cur.next = l2
                l2 = l2.next
            cur = cur.next
        if l1 != None:
            cur.next = l1
        if l2 != None:
            cur.next = l2
        return dummy.next
</pre>
Time Complexity: O(m+n), Space Complexity: O(1).

## 24. Swap Nodes in Pairs
<pre>
class Solution(object):
    
    def swapPairs(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        dummy = ListNode(0)
        pre = dummy
        pre.next = head
        while pre.next and pre.next.next:
            a = pre.next
            b = pre.next.next
            pre.next = b
            a.next = b.next
            b.next = a
            pre = a
        return dummy.next
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 61. Rotate List
<pre>
class Solution(object):
    
    def rotateRight(self, head, k):
        """
        :type head: ListNode
        :type k: int
        :rtype: ListNode
        """        
        buff = []
        while head != None:
            buff.append(head.val)
            head = head.next
        n = len(buff)
        if n == 0:
            return None
        else:
            k = k%n
        res = buff[n-k:] + buff[:n-k]
        pre = None
        while n != 0:
            temp = ListNode(res.pop())
            temp.next = pre
            pre = temp
            n -= 1
        return pre
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 82. Remove Duplicates from Sorted List II
<pre>
class Solution(object):
    
    def deleteDuplicates(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        buff = []
        dict = {}
        while head != None:
            buff.append(head.val)
            if head.val in dict:
                dict[head.val] += 1
            else:
                dict[head.val] = 1
            head = head.next
        res = []
        for key in buff:
            if dict[key] == 1:
                res.append(key)
        pre = None
        while len(res) != 0:
            cur = ListNode(res.pop())
            cur.next = pre
            pre = cur
        return pre
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 86. Partition List
<pre>
class Solution(object):
    
    def partition(self, head, x):
        """
        :type head: ListNode
        :type x: int
        :rtype: ListNode
        """
        small = preS = ListNode(0)
        large = preL = ListNode(0)
        while head != None:
            if head.val < x:
                preS.next = head
                preS = preS.next
            else:
                preL.next = head
                preL = preL.next
            head = head.next
        preL.next = None
        preS.next = large.next
        return small.next
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 141. Linked List Cycle
<pre>
class Solution(object):
    
    def hasCycle(self, head):
        """
        :type head: ListNode
        :rtype: bool
        """
        while head != None:
            if head.val == '#':
                return True
            else:
                head.val = '#'
                head = head.next
        return False
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 160. Intersection of Two Linked Lists
<pre>
class Solution(object):
    
    def getIntersectionNode(self, headA, headB):
        """
        :type head1, head1: ListNode
        :rtype: ListNode
        """
        if headA == None or headB == None:
            return None
        lenA = lenB = 1
        tailA = headA
        tailB = headB
        while tailA.next != None:
            lenA += 1
            tailA = tailA.next
        while tailB.next != None:
            lenB += 1
            tailB = tailB.next
        if tailA != tailB:
            return None
        if lenA > lenB:
            for i in range(lenA-lenB):
                headA = headA.next
        else:
            for i in range(lenB-lenA):
                headB = headB.next
        while headA != headB:
            headA = headA.next
            headB = headB.next
        return headA
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 203. Remove Linked List Elements
<pre>
class Solution(object):
    
    def removeElements(self, head, val):
        """
        :type head: ListNode
        :type val: int
        :rtype: ListNode
        """
        dummy = ListNode(0)
        dummy.next = head
        pre = dummy
        while pre.next != None:
            if pre.next.val == val:
                pre.next = pre.next.next
            else:
                pre = pre.next
        return dummy.next
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 206. Reverse Linked List

class Solution(object):
    
    def reverseList(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        pre = None
        while head != None:
            cur = head
            head = head.next
            cur.next = pre
            pre = cur
        return pre
</pre>
Time Complexity: O(n), Space Complexity: O(1).
